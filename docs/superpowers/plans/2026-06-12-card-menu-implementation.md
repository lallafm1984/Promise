# Card Menu Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the approved `언제볼래` card menu so users can create `이때볼래?` or `언제볼래?` cards from time/place only, preview/share them, and manage cards by status.

**Architecture:** Keep product rules in a pure helper module, render the card menu with focused React Native components, and keep `app/(tabs)/create.tsx` as the stateful screen container. Use the existing Expo Router, warm theme, UI primitives, mock repository, and current `PromiseCard` types.

**Tech Stack:** Expo SDK 56, React Native 0.85, React 19, Expo Router, TypeScript, Vitest, `expo-clipboard`, built-in React Native `Share`, `lucide-react-native`.

```ts
import type { AppointmentMode, PromiseCard } from '@/types/promise';

export interface CardDraft {
  mode: AppointmentMode;
  times: string[];
  location: string;
  message: string;
}

export type DraftValidationResult = { valid: true } | { valid: false; error: string };

export type ManagedStatusGroup = 'PENDING' | 'VOTING' | 'CONFIRMED' | 'PAST';

export type ManagedCardActionKind = 'RESHARE' | 'RESULTS' | 'SCHEDULE' | 'RECREATE';

export interface ManagedCardAction {
  kind: ManagedCardActionKind;
  label: string;
}

export function getModeLabel(mode: AppointmentMode) {
  return mode === 'DIRECT' ? '이때볼래?' : '언제볼래?';
}

export function compactDraftTimes(times: string[]) {
  return times.map((time) => time.trim()).filter(Boolean);
}

export function validateCardDraft(draft: CardDraft): DraftValidationResult {
  const times = compactDraftTimes(draft.times);

  if (times.length === 0) {
    return { valid: false, error: '만날 시간을 골라주세요.' };
  }

  if (!draft.location.trim()) {
    return { valid: false, error: '만날 장소를 적어주세요.' };
  }

  if (draft.mode === 'POLL' && times.length < 2) {
    return { valid: false, error: '언제볼래?는 후보 시간이 2개 이상 필요해요.' };
  }

  return { valid: true };
}

export function getGeneratedCardTitle(draft: CardDraft) {
  const times = compactDraftTimes(draft.times);
  const location = draft.location.trim();

  if (draft.mode === 'POLL') {
    return `${location}에서 언제볼래?`;
  }

  return `${times[0]}에 ${location}에서 볼래?`;
}

export function getPrimarySlot(card: PromiseCard) {
  return card.candidates.find((slot) => slot.id === card.selectedSlotId) ?? card.candidates[0];
}

export function getManagedStatusGroup(card: PromiseCard, now = new Date()): ManagedStatusGroup {
  const primarySlot = getPrimarySlot(card);

  if (card.status === 'DECLINED') {
    return 'PAST';
  }

  if (card.status === 'CONFIRMED' && primarySlot) {
    const startsAt = new Date(primarySlot.startsAt);
    if (!Number.isNaN(startsAt.getTime()) && startsAt.getTime() < now.getTime()) {
      return 'PAST';
    }
  }

  if (card.status === 'PENDING') {
    return 'PENDING';
  }

  if (card.status === 'VOTING') {
    return 'VOTING';
  }

  if (card.status === 'CONFIRMED') {
    return 'CONFIRMED';
  }

  return 'PAST';
}

export function getManagedCardAction(card: PromiseCard, now = new Date()): ManagedCardAction {
  const group = getManagedStatusGroup(card, now);

  if (group === 'PENDING') {
    return { kind: 'RESHARE', label: '공유 다시하기' };
  }

  if (group === 'VOTING') {
    return { kind: 'RESULTS', label: '결과 보기' };
  }

  if (group === 'CONFIRMED') {
    return { kind: 'SCHEDULE', label: '일정 보기' };
  }

  return { kind: 'RECREATE', label: '다시 만들기' };
}

export function buildShareMessage(card: PromiseCard) {
  const timeSummary = card.candidates.map((slot) => slot.label).join(' / ');
  return [
    card.title,
    `언제: ${timeSummary}`,
    `어디서: ${card.location}`,
    card.message.trim() ? `한마디: ${card.message.trim()}` : '',
    card.sharedUrl,
  ]
    .filter(Boolean)
    .join('\n');
}
```

export function ManagedCardsSection({ cards, onAction }: ManagedCardsSectionProps) {
  if (cards.length === 0) {
    return (
      <Card style={styles.emptyCard}>
        <Text style={styles.emptyTitle}>아직 만든 카드가 없어요</Text>
        <Text style={styles.emptyBody}>시간과 장소만 정하면 바로 공유할 수 있어요.</Text>

  return (

      {statusGroups.map((group) => {
        const groupCards = cards.filter((card) => getManagedStatusGroup(card) === group.key);

function ManagedCardRow({ card, onAction }: { card: PromiseCard; onAction: ManagedCardsSectionProps['onAction'] }) {

import { useMemo, useState } from 'react';

  CandidateTimeFields,

} from '@/components/card-menu';

  buildShareMessage,
  compactDraftTimes,
  getGeneratedCardTitle,
  getModeLabel,
  validateCardDraft,
  type CardDraft,
  type ManagedCardActionKind,
} from '@/lib/cardMenu';
import type { AppointmentMode, PromiseCard } from '@/types/promise';

const initialDraft: CardDraft = {
  mode: 'DIRECT',
  times: [''],
  location: '',
  message: '',
};

export default function CreateCardScreen() {
  const { profile, recentCards } = usePromiseData();
  const [draft, setDraft] = useState<CardDraft>(initialDraft);
  const [showMessage, setShowMessage] = useState(false);
  const [previewCard, setPreviewCard] = useState<PromiseCard | null>(null);
  const [createdCards, setCreatedCards] = useState<PromiseCard[]>([]);
  const [error, setError] = useState<string | null>(null);
  const [feedback, setFeedback] = useState<string | null>(null);

  const managedCards = useMemo(() => [...createdCards, ...recentCards], [createdCards, recentCards]);

  function updateMode(mode: AppointmentMode) {
    setError(null);
    setFeedback(null);
    setDraft((current) => ({
      ...current,
      mode,
      times: mode === 'POLL' && current.times.length < 2 ? [current.times[0] ?? '', ''] : [current.times[0] ?? ''],
    }));
  }

  function updateTime(index: number, value: string) {
    setDraft((current) => {
      const nextTimes = [...current.times];
      nextTimes[index] = value;
      return { ...current, times: nextTimes };
    });
  }

  function addCandidateTime() {
    setError(null);
    setFeedback(null);
    setDraft((current) => ({
      ...current,
      mode: 'POLL',
      times: [...current.times, ''],
    }));
  }

  function removeCandidateTime(index: number) {
    setError(null);
    setDraft((current) => {
      const nextTimes = current.times.filter((_, timeIndex) => timeIndex !== index);
      if (nextTimes.length <= 1) {
        setFeedback('후보 시간이 하나라 이때볼래?로 바뀌었어요.');
        return { ...current, mode: 'DIRECT', times: [nextTimes[0] ?? ''] };
      }

      return { ...current, times: nextTimes };
    });
  }

  function createPreview() {
    const validation = validateCardDraft(draft);
    setFeedback(null);

    if (!validation.valid) {
      setError(validation.error);
      return;
    }

    const card = buildPreviewCard(draft, profile?.displayName ?? '나');
    setPreviewCard(card);
    setError(null);
  }

  async function sharePreviewCard() {
    if (!previewCard) {
      return;
    }

    await Share.share({
      title: previewCard.title,
      message: buildShareMessage(previewCard),
    });
    publishPreviewCard(previewCard, '카톡 공유를 열었어요.');
  }

  async function copyPreviewLink() {
    if (!previewCard) {
      return;
    }

    await Clipboard.setStringAsync(buildShareMessage(previewCard));
    publishPreviewCard(previewCard, '공유 링크를 복사했어요.');
  }

  function publishPreviewCard(card: PromiseCard, message: string) {
    const publishedCard: PromiseCard = {
      ...card,
      status: card.mode === 'DIRECT' ? 'PENDING' : 'VOTING',
    };
    setCreatedCards((current) => [publishedCard, ...current.filter((item) => item.id !== publishedCard.id)]);
    setFeedback(message);
  }

  async function handleManagedAction(card: PromiseCard, action: ManagedCardActionKind) {
    if (action === 'RESHARE') {
      await Share.share({ title: card.title, message: buildShareMessage(card) });
      setFeedback('공유를 다시 열었어요.');
      return;
    }

    if (action === 'RECREATE') {
      setDraft({
        mode: card.mode,
        times: card.candidates.map((slot) => slot.label),
        location: card.location,
        message: card.message,
      });
      setShowMessage(Boolean(card.message.trim()));
      setPreviewCard(null);
      setFeedback('이전 카드 내용으로 다시 만들 수 있어요.');
      return;
    }

    if (action === 'RESULTS') {
      setFeedback('후보 시간별 응답을 확인하는 흐름으로 연결됩니다.');
      return;
    }

    setFeedback('확정된 시간과 장소를 확인하는 흐름으로 연결됩니다.');
  }

  return (
    <AppScreen>
      <View style={styles.header}>
        <Text style={styles.kicker}>언제볼래</Text>
        <Text style={styles.title}>카드 만들기</Text>
        <Text style={styles.subtitle}>시간과 장소만 정하면 바로 공유할 수 있어요.</Text>
      </View>

      <ModeSelector value={draft.mode} onChange={updateMode} />

      <Card style={styles.formCard}>
        <CandidateTimeFields
          mode={draft.mode}
          times={draft.times}
          onAdd={addCandidateTime}
          onChangeTime={updateTime}
          onRemove={removeCandidateTime}
        />
        <DraftInput
          label="어디서"
          value={draft.location}
          placeholder="예: 성수 카페"
          icon={<MapPin size={16} color={palette.primaryDeep} />}
          onChangeText={(location) => setDraft((current) => ({ ...current, location }))}
        />
        {showMessage ? (
          <DraftInput
            label="한마디"
            value={draft.message}
            placeholder="예: 늦으면 커피 내가 살게"
            icon={<MessageCircle size={16} color={palette.primaryDeep} />}
            multiline
            onChangeText={(message) => setDraft((current) => ({ ...current, message }))}
          />
        ) : (
          <ActionButton
            label="+ 한마디 추가"
            variant="secondary"
            icon={<MessageCircle size={17} color={palette.primaryDeep} />}
            onPress={() => setShowMessage(true)}
          />
        )}
      </Card>

      {error ? <Text style={styles.errorText}>{error}</Text> : null}
      {feedback ? <Text style={styles.feedbackText}>{feedback}</Text> : null}

      <ActionButton label="카드 만들기" icon={<Send size={18} color="#FFFFFF" />} fullWidth onPress={createPreview} />

      {previewCard ? (
        <View style={styles.previewSection}>
          <SectionHeader title="공유 전 미리보기" action={getModeLabel(previewCard.mode)} />
          <DraftPreviewCard card={previewCard} />
          <View style={styles.shareActions}>
            <ActionButton
              label="카톡 공유"
              variant="kakao"
              icon={<Send size={18} color={palette.ink} />}
              fullWidth
              onPress={sharePreviewCard}
            />
            <ActionButton
              label="링크 복사"
              variant="secondary"
              icon={<Link2 size={18} color={palette.primaryDeep} />}
              fullWidth
              onPress={copyPreviewLink}
            />
          </View>
        </View>
      ) : null}

      <View style={styles.managementSection}>
        <SectionHeader title="내 카드 관리함" />
        <ManagedCardsSection cards={managedCards} onAction={handleManagedAction} />
      </View>
    </AppScreen>
  );
}

function buildPreviewCard(draft: CardDraft, hostName: string): PromiseCard {
